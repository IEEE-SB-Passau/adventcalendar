Hier findest du eine Reihe von Beispielprogrammen, welche anhand einer einfachen Aufgabe demonstrieren wie die Ein- und Ausgabe implementiert werden kann.

Aufgabe

Deine Aufgabe ist es, ein Programm zu entwickeln, welches eine Liste von Zahlen einliest, die Quersumme jeder Zahl berechnet, und diese dann ausgibt.

Eingabe

Die erste Zeile der Eingabe beschreibt, wie oft eine Quersumme bestimmt werden muss (1 ≤ n ≤ 1000). Nach der ersten Zeile folgen n weitere Zeilen. Jede dieser Zeilen enthält ausschließlich Ziffern. Die Anzahl der Ziffern pro Zeile ist auf (1 ≤ z ≤ 50) begrenzt.

Ausgabe

Zu jeder eingegebenen Zeile muss die entsprechende Quersumme ausgegeben werden. Jede Quersumme muss auf einer eigenen Zeile ausgegeben werden: Es muss also nach jeder Quersumme, auch der letzten, ein Newline (\n oder \r\n) ausgegeben werden.

Beispiel

Eingabe
5
00001
1234
0101
5005
22
Ausgabe
1
10
2
10
4

Implementierungen

Java

import java.util.Scanner;

public class Solution {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int numTestcases = Integer.parseInt(in.nextLine());

        for (int i = 0; i < numTestcases; i++) {
            String testcase = in.nextLine();
            int result = 0;
            for (int j = 0; j < testcase.length(); j++) {
                char c = testcase.charAt(j);
                result += Character.getNumericValue(c);
            }
            System.out.println(result);
        }
     }

}

JavaScript

var readline = require("readline");
var rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

function solve(testcase) {
    var sum = 0;
    for (var i = 0; i < testcase.length; i++) {
        sum += parseInt(testcase.charAt(i));
    }
    return sum;
}

var numTestcases = null;
var done = 0;

rl.on("line", function(line) {
    if (numTestcases === null) {
        numTestcases = parseInt(line);
    } else if (done < numTestcases) {
        var solution = solve(line);
        process.stdout.write(solution + "\n");
        done++;
    }
});

#include <stdio.h>

int main(int argv, int** argc) {
    int num_testcases;
    fscanf(stdin,"%d\n", &num_testcases);

    for(int i = 0; i < num_testcases; i++) {
        int sum = 0;
        int j;

        while((j = fgetc(stdin)) != '\n') {
            if(j == EOF) return 1;
            // note: ascii values for '0'-'9' are 48-57
            sum += j - 48;
        }

        printf("%d\n", sum);
    }
}

Haskell

--
-- Quersummen-Aufgabe
--
-- Aufruf mittels
--    ghc -e main QSum1.hs
-- ("-e main" ruft die Funktion "main" als Programmeintrittspunkt auf).
--
-- Alternativ: Programm kompilieren mit
--    ghc -o qsum1 QSum1.hs
-- und dann mittels
--    ./qsum1
-- das Kompilat starten.
--

--
-- Loesung fuer die Quersummen-Aufgabe.
--
solution :: String -> String
solution input = output
    where
      -- Teile die gesamte Eingabe (input) in die einzelnen Zeilen
      -- auf mit Hilfe der Funktion lines :: String -> [String].
      inputLines :: [String]
      inputLines = lines input

      -- "head inputLines" ist die erste eingegebene Zahl (= Anzahl
      -- der Quersummenberechnungen, die durchgefuehrt werden sollen).
      -- Mittels "read" wird aus dem String ein Int.
      count :: Int
      count = read (head inputLines)

      -- Die Zahlen, fuer die die Quersummen berechnet werden sollen,
      -- folgen der Anzahl (daher "tail inputLines"). Davon nehmen wir
      -- "count" viele und wandeln alle Strings in Integer um
      -- (mittels "read").
      numbers :: [Integer]
      numbers = map read (take count (tail inputLines))

      -- Berechne fuer jede der eingegebenen Zahlen die Quersumme.
      qsums :: [Integer]
      qsums = map qsum numbers

      -- Wandle (mittels "show") die berechneten Quersummen in Strings um.
      outputLines :: [String]
      outputLines = map show qsums

      -- Setze die Ausgabe-Strings mittels "unlines :: [String] -> String"
      -- zur Ausgabe zusammen. unlines fuegt nach jedem String ein
      -- Newline ("\n") hinzu, auch nach dem letzten String.
      output :: String
      output = unlines outputLines

-- Klassische (nicht-iterierte) Quersummenberechnung
-- fuer nicht-negative Zahlen.
qsum :: Integer -> Integer
qsum x
    | x < 10     =  x
    | otherwise  =  (x `mod` 10) + qsum (x `div` 10)

--
-- Programmeintrittspunkt: "main"
-- Die Funktion "main" muss die Signatur "IO ()" haben.
-- Die Funktion "interact :: (String -> String) -> IO ()"
-- nimmt als Argument (hier: "solution") eine Funktion (von String
-- nach String), an die dann die gesamte Eingabe (als String) uebergeben
-- wird und die die Ausgabe (ebenfalls als String) produziert.
--
main :: IO ()
main = interact solution

C#

using System;

public class Solution {

    public static void Main(string[] args) {
        int numTc;
        // Read a line from STDIN
        string input = Console.ReadLine();

        /*
        * Parse the number of test cases by providing its textual
        * representation and the target int variable (the latter as an
        * output parameter);
        *
        * Instead of throwing an exception, int.TryParse returns false if
        * the given input string is not an integer value.
        */
        if (!int.TryParse(input, out numTc) || numTc < 1 || numTc > 1000) {
            // Write error message to STDERR
            Console.Error.WriteLine("Malformed testcase count!");
            return;
        }

        for (int i = 0; i < numTc; ++i) {
            string testCase = Console.ReadLine();
            // Compute result and write to STDOUT
            int sum = computeSum(testCase);
            Console.WriteLine(sum);
        }
    }

    private static int computeSum(string digitString) {
        int sum = 0;
        foreach (char c in digitString) {
            if (c >= '0' && c <= '9') {
                sum += c - '0';
            } else {
                throw new ArgumentException("Not a digit!");
            }
        }
        return sum;
    }

}
